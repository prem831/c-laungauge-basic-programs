#include<stdio.h>

       void str_cat(char *d,char *s)
	  {
	     int i,j;
	     i=0;
	     while(d[i]!='\0')
		i++;
	     j=0;
	     while(s[j]!='\0')
	     {
	     d[i]=s[j];
	     i++;
	     j++;
	     }
	     d[i]='\0';
	     return;
	  }
